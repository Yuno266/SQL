create table salary(
id serial  primary key,
monthly_salary int not null
);



create table roles(
id serial primary key,
role_title varchar(50) unique not null
);

create table roles_salary(
id serial primary key,
role_id int not null,
salary_id int not null,
foreign key (role_id)
	references roles(id),
foreign key (salary_id)
	references salary(id)
);
